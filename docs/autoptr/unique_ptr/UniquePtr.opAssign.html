    Function UniquePtr.opAssign

    Releases the ownership of the managed object, if any.

    void opAssign(MemoryOrder order = MemoryOrder.seq, This) (
      typeof(null) nil
    ) scope
    if (isMutable!This);

    After the call, this manages no object.

    Examples

    {
    UniquePtr!long x = UniquePtr!long.make(1);

    assert(x != null);
    assert(*x == 1);
    x = null;
    assert(x == null);
}

{
    UniquePtr!(shared long) x = UniquePtr!(shared long).make(1);

    assert(x != null);
    assert(*x == 1);
    x = null;
    assert(x == null);
}

    Function UniquePtr.opAssign

    Move ownership of the object managed by rhs.

    void opAssign(MemoryOrder order = MemoryOrder.seq, Rhs, This) (
      scope Rhs rhs
    ) scope @safe
    if (true && isUniquePtr!Rhs && isAssignable!(Rhs, This) && !is(Rhs == shared) && isMutable!This);

    If rhs manages no object, this manages no object too.

    Examples

    import core.lifetime : move;
{
    UniquePtr!long px1 = UniquePtr!long.make(1);
    UniquePtr!long px2 = UniquePtr!long.make(2);

    px1 = move(px2);
    assert(px2 == null);
    assert(px1 != null);
    assert(*px1 == 2);
}


{
    UniquePtr!long px = UniquePtr!long.make(1);
    UniquePtr!(const long) pcx = UniquePtr!long.make(2);

    pcx = move(px);
    assert(px == null);
    assert(pcx != null);
    assert(*pcx == 1);

}
