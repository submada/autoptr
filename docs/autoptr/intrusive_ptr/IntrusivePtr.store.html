    Function IntrusivePtr.store

    Stores the non shared IntrusivePtr parameter ptr to this.

    void store(MemoryOrder order = MemoryOrder.seq, This) (
      typeof(null) nil
    ) scope
    if (isMutable!This);

    void store(MemoryOrder order = MemoryOrder.seq, Ptr, This) (
      scope ref Ptr desired
    ) scope
    if (true && isIntrusivePtr!Ptr && !is(Ptr == shared) && !needLock!(Ptr, This) && (isCopyable!(Ptr, This) && isMutable!This));

    void store(MemoryOrder order = MemoryOrder.seq, Ptr, This) (
      scope Ptr desired
    ) scope
    if (true && isIntrusivePtr!Ptr && !is(Ptr == shared) && !needLock!(Ptr, This) && (isMovable!(Ptr, This) && isMutable!This));

    If this is shared then operation is atomic or guarded by mutex.

    Template parameter order has type core.atomic.MemoryOrder.

    Examples

    //null store:
{
    shared x = IntrusivePtr!(shared long).make(123);
    assert(x.load.get == 123);

    x.store(null);
    assert(x.useCount == 0);
    assert(x.load == null);
}

//rvalue store:
{
    shared x = IntrusivePtr!(shared long).make(123);
    assert(x.load.get == 123);

    x.store(IntrusivePtr!(shared long).make(42));
    assert(x.load.get == 42);
}

//lvalue store:
{
    shared x = IntrusivePtr!(shared long).make(123);
    auto y = IntrusivePtr!(shared long).make(42);

    assert(x.load.get == 123);
    assert(y.load.get == 42);

    x.store(y);
    assert(x.load.get == 42);
    assert(x.useCount == 2);
}
