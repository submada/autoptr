    Same as SharedPtr.make but DestructorType is inferred.

    Example

    enum bool supportGC = true;

//
{
    auto s = makeSharedPtr!long(42);
}

{
    auto s = makeSharedPtr!(long, supportGC)(42);
}

{
    auto s = makeSharedPtr!(long, shared(SharedControlType), supportGC)(42);
}

// dynamic array:
{
    auto s = makeSharedPtr!(long[])(10, 42);
    assert(s.length == 10);
}
{
    auto s = makeSharedPtr!(long[], supportGC)(10, 42);
    assert(s.length == 10);
}
{
    auto s = makeSharedPtr!(long[], shared(SharedControlType), supportGC)(10, 42);
    assert(s.length == 10);
}


// deleter
long x = 42;
static void deleter(long* var)nothrow{
    (*var) += 1;
}

{
    auto s = makeSharedPtr!long(&x, &deleter);
}
assert(x == 43);

{
    auto s = makeSharedPtr!(long, supportGC)(&x, &deleter);
}
assert(x == 44);

{
    auto s = makeSharedPtr!(long, shared(SharedControlType), supportGC)(&x, &deleter);
}
assert(x == 45);
